  Database Scripts

  • DB Creation and users
    • mysql --host=localhost --user=root --password=xxxxxxxx --show-warnings
      • CREATE DATABASE NODE_WEB_TEMPLATE DEFAULT CHARSET=latin1 COLLATE=latin1_general_ci;
      • CREATE USER 'nwtemplate_dba'@'%' IDENTIFIED BY 'nwtemplate_dba';
      • CREATE USER 'nwtemplate_usr'@'%' IDENTIFIED BY 'nwtemplate_usr';
      • GRANT ALL PRIVILEGES ON NODE_WEB_TEMPLATE.* TO 'nwtemplate_dba'@'%';
      • GRANT SELECT, INSERT, UPDATE, DELETE ON NODE_WEB_TEMPLATE.* TO 'nwtemplate_usr'@'%';
      • FLUSH PRIVILEGES;
    • QUIT;
  • Objects Creation (Tables, indexes, etc. will belong to the owner 'nwtemplate_dba' instead of 'root')
    • mysql --host=localhost --user=nwtemplate_dba --password=nwtemplate_dba NODE_WEB_TEMPLATE --show-warnings;
      • Table USER
        • CREATE TABLE NODE_WEB_TEMPLATE.USER ( USER_ID INT(10) UNSIGNED NOT NULL AUTO_INCREMENT , PRIMARY KEY (USER_ID), EMAIL VARCHAR(200) NOT NULL, FIRST_NAME VARCHAR(50) NOT NULL, LAST_NAME VARCHAR(50) NOT NULL, PASSWORD VARCHAR(8) NOT NULL, STATUS CHAR(1) NOT NULL ) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_general_ci;
        • In a real case scenario, the password field must be long enough to store an encrypted password
      • Table USER_SESSIONS
        • CREATE TABLE NODE_WEB_TEMPLATE.USER_SESSIONS ( USER_ID INT(10) UNSIGNED NOT NULL, DEVICE_ID TINYINT NOT NULL, AUTH_TOKEN VARCHAR(256) NOT NULL, TMSTP_LOGIN DATETIME NOT NULL ) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_general_ci;
        • ALTER TABLE NODE_WEB_TEMPLATE.USER_SESSIONS ADD PRIMARY KEY PK_USER_SESSIONS (USER_ID, DEVICE_ID);
      • Table USER_DEVICES
        • CREATE TABLE NODE_WEB_TEMPLATE.USER_DEVICES ( USER_ID INT(10) UNSIGNED NOT NULL, DEVICE_ID TINYINT NOT NULL, DEVICE_DESCRIPTION VARCHAR(50) NOT NULL, TMSTP_CREATION DATETIME NOT NULL ) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_general_ci;
        • ALTER TABLE NODE_WEB_TEMPLATE.USER_DEVICES ADD PRIMARY KEY PK_USER_DEVICES (USER_ID, DEVICE_ID);
    • QUIT;
